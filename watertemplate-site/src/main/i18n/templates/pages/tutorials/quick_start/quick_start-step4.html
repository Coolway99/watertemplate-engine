    class MonthsGrid extends Template {

        private static final Collection months = Arrays.asList(Month.values());

        MonthsGrid(final Year year) {
            add("year", year.toString());

            addCollection("months", months, (month, map) -> {
                map.add("lower_name", month.name().toLowerCase());
                map.add("days_count", month.length(year.isLeap()) + "");
            });
        }

        @Override
        protected String getFilePath() {
            return "months_grid.html";
        }
    }
